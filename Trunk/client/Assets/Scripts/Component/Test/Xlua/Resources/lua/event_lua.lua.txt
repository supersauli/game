Event = require "Xlua/3rd/Event/events"

show = function()
	print("show xxxx")
end
Event.AddListener("Show",show)

Event.Brocast("Show")
Event.RemoveListener("Show")

Event.Brocast("Show")

Event.AddListener("Show2",
function()
	print("Show2")
end
)
Event.Brocast("Show2")
Event.RemoveListener("Show2",
function()
	print("Show2")
end
)

Event.Brocast("Show")
Event.Brocast("Show2")
